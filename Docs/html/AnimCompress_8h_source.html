26// Logic for tracking top N error items for later display
27template<typename DataType, typename SortType, int MaxItems>
28struct FMaxErrorStatTracker
29{
30public:
31 FMaxErrorStatTracker()
32 : CurrentLowestError(0.f)
33 {
34 Items.Reserve(MaxItems);
35 }
36
37 bool CanUseErrorStat(SortType NewError)
38 {
39 return Items.Num() < MaxItems || NewError > CurrentLowestError;
40 }
41
42 template <typename... ArgsType>
43 void StoreErrorStat(SortType NewError, ArgsType&&... Args)
44 {
45 bool bModified = false;
46
47 if (Items.Num() < MaxItems)
48 {
49 Items.Emplace(Forward<ArgsType>(Args)...);
50 bModified = true;
51 }
52 else if(NewError > CurrentLowestError)
53 {
54 Items[MaxItems - 1] = DataType(Forward<ArgsType>(Args)...);
55 bModified = true;
56 }
57
58 if (bModified)
59 {
60 Algo::Sort(Items, TGreater<>());
61 CurrentLowestError = Items.Last().GetErrorValue();
62 }
63 }
64
65 void LogErrorStat()
66 {
67 for (int ItemIndex = 0; ItemIndex < Items.Num(); ++ItemIndex)
68 {
69 UE_LOG(LogAnimationCompression, Display, TEXT("%i) %s"), ItemIndex+1, *Items[ItemIndex].ToText().ToString());
70 }
71 }
72
73 const DataType& GetMaxErrorItem() const
74 {
75 return Items[0];
76 }
77
78 bool IsValid() const
79 {
80 return Items.Num() > 0;
81 }
82
83private:
84 //Storage of tracked items
85 TArray<DataType> Items;
86
87 //For ease cache current lowest error value
88 SortType CurrentLowestError;
89};
