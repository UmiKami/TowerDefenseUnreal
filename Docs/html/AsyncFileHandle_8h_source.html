47 FORCEINLINE IAsyncReadRequest(FAsyncFileCallBack* InCallback, bool bInSizeRequest, uint8* UserSuppliedMemory)
48 : bDataIsReady(false)
49 , bCompleteAndCallbackCalled(false)
50 , bCompleteSync(false)
51 , bCanceled(false)
52 , bSizeRequest(bInSizeRequest)
53 , bUserSuppliedMemory(!!UserSuppliedMemory)
54 {
55 if (InCallback)
56 {
57 Callback = *InCallback;
58 }
59 if (bSizeRequest)
60 {
61 Size = -1;
62 check(!UserSuppliedMemory && !bUserSuppliedMemory); // size requests don't do memory
63 }
64 else
65 {
66 Memory = UserSuppliedMemory;
67 check((!!Memory) == bUserSuppliedMemory);
68 }

70 }

72 /* Not legal to destroy the request until it is complete. */
73 virtual ~IAsyncReadRequest()
74 {
75 // check(bCompleteAndCallbackCalled && (bSizeRequest || !Memory)); // must be complete, and if it was a read request, the memory should be gone
76 UE_CLOG(!(bCompleteAndCallbackCalled && (bSizeRequest || !Memory)),
77 LogHAL, Fatal, TEXT("IAsyncReadRequests must not be deleted until they are completed.")

86 FORCEINLINE bool PollCompletion()
87 {
88 return bCompleteAndCallbackCalled;
89 }
90
96 FORCEINLINE bool WaitCompletion(float TimeLimitSeconds = 0.0f)
97 {
98 if (PollCompletion())
99 {
100 return true;
101 }
102 WaitCompletionImpl(TimeLimitSeconds);
103 return PollCompletion();
104 }
105
110 virtual void EnsureCompletion()
111 {
112 // Default implementation is the same as WaitCompletion(0.0f) except that it skips the testing of
113 // PollCompletion. This is potentially slower because we do not early exit if PollCompletion is true, but it
114 // provides a stronger guarantee of completion because PollCompletion can sometimes return true while
115 // completion steps are still in progress.
116 WaitCompletionImpl(0.0f);
117 }
118
120 FORCEINLINE void Cancel()
121 {
122 if (!bCanceled)
123 {
124 bCanceled = true;
125 bDataIsReady = true;
126 FPlatformMisc::MemoryBarrier();
127 if (!PollCompletion())
128 {
129 return CancelImpl();
130 }
131 }
132 }
133
138 FORCEINLINE int64 GetSizeResults()
139 {
140 check(bDataIsReady && bSizeRequest);
141 return bCanceled ? -1 : Size;
142 }
143
148 FORCEINLINE uint8* GetReadResults()
149 {
150 check(bDataIsReady && !bSizeRequest);
151 uint8* Result = Memory;
152 if (bCanceled && Result)
153 {
154 Result = nullptr;
155 }
156 else
157 {
158 if (Memory && !bUserSuppliedMemory)
159 {
160 ReleaseMemoryOwnershipImpl();
161 }
162
163 Memory = nullptr;
164 }
165 return Result;
166 }
167
168protected:
174 virtual void WaitCompletionImpl(float TimeLimitSeconds) = 0;
175
177 virtual void CancelImpl() = 0;
178
184 virtual void ReleaseMemoryOwnershipImpl() = 0;
185
186 void SetDataComplete()
187 {
188 bDataIsReady = true;
189 FPlatformMisc::MemoryBarrier();
190 if (Callback)
191 {
192 Callback(bCanceled, this);
193 }
194 FPlatformMisc::MemoryBarrier();
195 }
196
197 void SetAllComplete()
198 {
199 bCompleteAndCallbackCalled = true;
200 FPlatformMisc::MemoryBarrier();
201 }
202
203 void SetComplete()
204 {
205 SetDataComplete();
206 SetAllComplete();
207 }
