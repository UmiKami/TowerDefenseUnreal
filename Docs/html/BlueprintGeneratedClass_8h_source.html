126template<typename ElementType>
127class TSimpleRingBuffer
128{
129public:
130 TSimpleRingBuffer(int32 MaxItems)
131 : WriteIndex(0)
132 {
133 Storage.Empty(MaxItems);
134 }
135
136 int32 Num() const
137 {
138 return Storage.Num();
139 }
140
141 ElementType& operator()(int32 i)
142 {
143 // First element is at WriteIndex-1, second at 2, etc...
144 const int32 RingIndex = (WriteIndex - 1 - i);
145 const int32 WrappedRingIndex = (RingIndex < 0) ? (RingIndex + ArrayMax()) : RingIndex;
146
147 return Storage[WrappedRingIndex];
148 }
149
150 const ElementType& operator()(int32 i) const
151 {
152 // First element is at WriteIndex-1, second at 2, etc...
153 const int32 RingIndex = (WriteIndex - 1 - i);
154 const int32 WrappedRingIndex = (RingIndex < 0) ? (RingIndex + ArrayMax()) : RingIndex;
155
156 return Storage[WrappedRingIndex];
157 }
158
159 int32 ArrayMax() const
160 {
161 return Storage.GetSlack() + Storage.Num();
162 }
163
164 // Note: Doesn't call the constructor on objects stored in it; make sure to properly initialize the memory returned by WriteNewElementUninitialized
165 ElementType& WriteNewElementUninitialized()
166 {
167 const int32 OldWriteIndex = WriteIndex;
168 WriteIndex = (WriteIndex + 1) % ArrayMax();
169
170 if (Storage.GetSlack())
171 {
172 Storage.AddUninitialized(1);
173 }
174 return Storage[OldWriteIndex];
175 }
176
177 ElementType& WriteNewElementInitialized()
178 {
179 const int32 OldWriteIndex = WriteIndex;
180 WriteIndex = (WriteIndex + 1) % ArrayMax();
181
182 if (Storage.GetSlack())
183 {
184 Storage.Add(ElementType());
185 }
186 else
187 {
188 Storage[OldWriteIndex] = ElementType();
189 }
190 return Storage[OldWriteIndex];
191 }
192
193private:
194 TArray<ElementType> Storage;
195 int32 WriteIndex;
196};
