547FORCEINLINE TVector2<T> TBox2<T>::GetClosestPointTo( const TVector2<T>& Point ) const
548{
549 // start by considering the point inside the box
550 TVector2<T> ClosestPoint = Point;
551
552 // now clamp to inside box if it's outside
553 if (Point.X < Min.X)
554 {
555 ClosestPoint.X = Min.X;
556 }
557 else if (Point.X > Max.X)
558 {
559 ClosestPoint.X = Max.X;
560 }
561
562 // now clamp to inside box if it's outside
563 if (Point.Y < Min.Y)
564 {
565 ClosestPoint.Y = Min.Y;
566 }
567 else if (Point.Y > Max.Y)
568 {
569 ClosestPoint.Y = Max.Y;
570 }

575template<typename T>
576TBox2<T> TBox2<T>::Overlap(const TBox2<T>& Other) const
577{
578 if (Intersect(Other) == false)
579 {
580 static TBox2<T> EmptyBox(ForceInit);
581 return EmptyBox;
582 }
583
584 // otherwise they overlap
585 // so find overlapping box
586 TVector2<T> MinVector, MaxVector;
587
588 MinVector.X = FMath::Max(Min.X, Other.Min.X);
589 MaxVector.X = FMath::Min(Max.X, Other.Max.X);
590
591 MinVector.Y = FMath::Max(Min.Y, Other.Min.Y);
592 MaxVector.Y = FMath::Min(Max.Y, Other.Max.Y);
593
594 return TBox2<T>(MinVector, MaxVector);
595}
