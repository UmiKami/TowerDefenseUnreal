714template<typename T>
715FORCEINLINE TBox<T>& TBox<T>::operator+=( const TVector<T> &Other )
716{
717 if (IsValid)
718 {
719 DiagnosticCheck();
720 // Not testing Other.DiagnosticCheck() since TVector is not initialized to NaN
721
722 Min.X = FMath::Min(Min.X, Other.X);
723 Min.Y = FMath::Min(Min.Y, Other.Y);
724 Min.Z = FMath::Min(Min.Z, Other.Z);
725
726 Max.X = FMath::Max(Max.X, Other.X);
727 Max.Y = FMath::Max(Max.Y, Other.Y);
728 Max.Z = FMath::Max(Max.Z, Other.Z);
729 }
730 else
731 {
732 Min = Max = Other;
733 IsValid = 1;
734 }
735
736 return *this;
737}

765template<typename T>
766FORCEINLINE TVector<T> TBox<T>::GetClosestPointTo( const TVector<T>& Point ) const
767{
768 DiagnosticCheck();
769
770 // start by considering the point inside the box
771 TVector<T> ClosestPoint = Point;
772
773 // now clamp to inside box if it's outside
774 if (Point.X < Min.X)
775 {
776 ClosestPoint.X = Min.X;
777 }
778 else if (Point.X > Max.X)
779 {
780 ClosestPoint.X = Max.X;
781 }
782
783 // now clamp to inside box if it's outside
784 if (Point.Y < Min.Y)
785 {
786 ClosestPoint.Y = Min.Y;
787 }
788 else if (Point.Y > Max.Y)
789 {
790 ClosestPoint.Y = Max.Y;
791 }
792
793 // Now clamp to inside box if it's outside.
794 if (Point.Z < Min.Z)
795 {
796 ClosestPoint.Z = Min.Z;
797 }
798 else if (Point.Z > Max.Z)
799 {
800 ClosestPoint.Z = Max.Z;
801 }
802
803 return ClosestPoint;
804}

867template<typename T>
868TBox<T> TBox<T>::TransformBy(const TMatrix<T>& M) const
869{
870 // if we are not valid, return another invalid box.
871 if (!IsValid)
872 {
873 return TBox<T>(ForceInit);
874 }
875
876 DiagnosticCheck();
877
878 TBox<T> NewBox;
879
880 const TVectorRegisterType<T> VecMin = VectorLoadFloat3_W0(&Min);
881 const TVectorRegisterType<T> VecMax = VectorLoadFloat3_W0(&Max);
882
883 const TVectorRegisterType<T> m0 = VectorLoadAligned(M.M[0]);
884 const TVectorRegisterType<T> m1 = VectorLoadAligned(M.M[1]);
885 const TVectorRegisterType<T> m2 = VectorLoadAligned(M.M[2]);
886 const TVectorRegisterType<T> m3 = VectorLoadAligned(M.M[3]);
887
888 const TVectorRegisterType<T> Half = VectorSetFloat1((T)0.5f); // VectorSetFloat1() can be faster than SetFloat3(0.5, 0.5, 0.5, 0.0). Okay if 4th element is 0.5, it's multiplied by 0.0 below and we discard W anyway.
889 const TVectorRegisterType<T> Origin = VectorMultiply(VectorAdd(VecMax, VecMin), Half);
890 const TVectorRegisterType<T> Extent = VectorMultiply(VectorSubtract(VecMax, VecMin), Half);
891
892 TVectorRegisterType<T> NewOrigin = VectorMultiply(VectorReplicate(Origin, 0), m0);
893 NewOrigin = VectorMultiplyAdd(VectorReplicate(Origin, 1), m1, NewOrigin);
894 NewOrigin = VectorMultiplyAdd(VectorReplicate(Origin, 2), m2, NewOrigin);
895 NewOrigin = VectorAdd(NewOrigin, m3);
896
897 TVectorRegisterType<T> NewExtent = VectorAbs(VectorMultiply(VectorReplicate(Extent, 0), m0));
898 NewExtent = VectorAdd(NewExtent, VectorAbs(VectorMultiply(VectorReplicate(Extent, 1), m1)));
899 NewExtent = VectorAdd(NewExtent, VectorAbs(VectorMultiply(VectorReplicate(Extent, 2), m2)));
900
901 const TVectorRegisterType<T> NewVecMin = VectorSubtract(NewOrigin, NewExtent);
902 const TVectorRegisterType<T> NewVecMax = VectorAdd(NewOrigin, NewExtent);
903
904 VectorStoreFloat3(NewVecMin, &(NewBox.Min.X));
905 VectorStoreFloat3(NewVecMax, &(NewBox.Max.X));
906
907 NewBox.IsValid = 1;
908
909 return NewBox;
910}

933template<typename T>
934TBox<T> TBox<T>::InverseTransformBy(const TTransform<T>& M) const
935{
936 TVector<T> Vertices[8];
937 GetVertices(Vertices);
938
939 TBox<T> NewBox(ForceInit);
940
941 for (int32 VertexIndex = 0; VertexIndex < UE_ARRAY_COUNT(Vertices); VertexIndex++)
942 {
943 TVector<T> ProjectedVertex = M.InverseTransformPosition(Vertices[VertexIndex]);
944 NewBox += ProjectedVertex;
945 }
946

950template<typename T>
951TBox<T> TBox<T>::TransformProjectBy(const TMatrix<T>& ProjM) const
952{
953 TVector<T> Vertices[8];
954 GetVertices(Vertices);
955
956 TBox<T> NewBox(ForceInit);
957
958 for (int32 VertexIndex = 0; VertexIndex < UE_ARRAY_COUNT(Vertices); VertexIndex++)
959 {
960 TVector4<T> ProjectedVertex = ProjM.TransformPosition(Vertices[VertexIndex]);
961 NewBox += ((TVector<T>)ProjectedVertex) / ProjectedVertex.W;
962 }
963
964 return NewBox;

967template<typename T>
968TBox<T> TBox<T>::Overlap(const TBox<T>& Other) const
969{
970 if (Intersect(Other) == false)
971 {
972 static TBox<T> EmptyBox(ForceInit);
973 return EmptyBox;
974 }
975
976 // otherwise they overlap
977 // so find overlapping box
978 TVector<T> MinVector, MaxVector;
979
980 MinVector.X = FMath::Max(Min.X, Other.Min.X);
981 MaxVector.X = FMath::Min(Max.X, Other.Max.X);
982
983 MinVector.Y = FMath::Max(Min.Y, Other.Min.Y);
984 MaxVector.Y = FMath::Min(Max.Y, Other.Max.Y);
985
986 MinVector.Z = FMath::Max(Min.Z, Other.Min.Z);
987 MaxVector.Z = FMath::Min(Max.Z, Other.Max.Z);
988
989 return TBox<T>(MinVector, MaxVector);
990}

1191 DistSquared += FMath::Square(SphereCenter.Y - AABB.Min.Y);
