23inline FVector3d xyYToXYZ(const FVector3d& xyY)
24{
25 double Divisor = FMath::Max(xyY[1], 1e-10);
26
27 return FVector3d(
28 xyY[0] * xyY[2] / Divisor,
29 xyY[2],
30 (1.0 - xyY[0] - xyY[1]) * xyY[2] / Divisor
31 );
32}

39inline FVector3d XYZToxyY(const FVector3d& XYZ)
40{
41 double Divisor = XYZ[0] + XYZ[1] + XYZ[2];
42 if (Divisor == 0.0)
43 {
44 Divisor = 1e-10;
45 }
46
47 return FVector3d(
48 XYZ[0] / Divisor,
49 XYZ[1] / Divisor,
50 XYZ[1]
51 );
52}
