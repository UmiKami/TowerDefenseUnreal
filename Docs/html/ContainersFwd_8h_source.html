19// FWideString is an alias for FString.
20//
21// This is so ANSICHAR/UTF8CHAR/WIDECHAR can be matched with FAnsiString/FUtf8String/FWideString when overloaded or specialized.
22//
23// FWideString should be the 'real' string class and FString should be the alias, but can't be for legacy reasons.
24// Forward declarations of FString expect it to be a class and changing it would affect ABIs and bloat PDBs.
25using FWideString = FString;
