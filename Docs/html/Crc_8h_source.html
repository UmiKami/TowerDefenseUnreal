44 template <typename CharType>
45 [[nodiscard]] static uint32 StrCrc32(const CharType* Data, uint32 CRC = 0)
46 {
47 // We ensure that we never try to do a StrCrc32 with a CharType of more than 4 bytes. This is because
48 // we always want to treat every CRC as if it was based on 4 byte chars, even if it's less, because we
49 // want consistency between equivalent strings with different character types.
50 static_assert(sizeof(CharType) <= 4, "StrCrc32 only works with CharType up to 32 bits.");
51
52 if constexpr (sizeof(CharType) != 1)
53 {
54 CRC = ~CRC;
55 while (CharType Ch = *Data++)
56 {
57 CRC = (CRC >> 8) ^ CRCTablesSB8[0][(CRC ^ Ch) & 0xFF];
58 Ch >>= 8;
59 CRC = (CRC >> 8) ^ CRCTablesSB8[0][(CRC ^ Ch) & 0xFF];
60 Ch >>= 8;
61 CRC = (CRC >> 8) ^ CRCTablesSB8[0][(CRC ^ Ch) & 0xFF];
62 Ch >>= 8;
63 CRC = (CRC >> 8) ^ CRCTablesSB8[0][(CRC ^ Ch) & 0xFF];
64 }
65 return ~CRC;
66 }
67 else
68 {
69 /* Special case for when CharType is a byte, which causes warnings when right-shifting by 8 */
70 CRC = ~CRC;
71 while (CharType Ch = *Data++)
72 {
73 CRC = (CRC >> 8) ^ CRCTablesSB8[0][(CRC ^ Ch) & 0xFF];
74 CRC = (CRC >> 8) ^ CRCTablesSB8[0][(CRC ) & 0xFF];
75 CRC = (CRC >> 8) ^ CRCTablesSB8[0][(CRC ) & 0xFF];
76 CRC = (CRC >> 8) ^ CRCTablesSB8[0][(CRC ) & 0xFF];
77 }
78 return ~CRC;
79 }
80 }

135 template <typename CharType>
136 static inline uint32 StrCrc_DEPRECATED(const CharType* Data)
137 {
138 // make sure table is initialized
139 check(CRCTable_DEPRECATED[1] != 0);
140
141 uint32 CRC = 0xFFFFFFFF;
142 while (*Data)
143 {
144 CharType C = *Data++;
145 int32 CL = (C&255);
146 CRC = (CRC << 8) ^ CRCTable_DEPRECATED[(CRC >> 24) ^ CL];

148 CRC = (CRC << 8) ^ CRCTable_DEPRECATED[(CRC >> 24) ^ CH];
149 }
150 return ~CRC;

248template <>
249inline uint32 FCrc::Strihash_DEPRECATED(const UTF8CHAR* Data)
250{
251 // We can't utilize StringConv.h here due to circular includes, so do the conversion manually
252
253 int32 Len = FPlatformString::Strlen(Data);
254 int32 ConvertedLen = FPlatformString::ConvertedLength<WIDECHAR>(Data, Len);
255
256 WIDECHAR* Temp = new WIDECHAR[ConvertedLen];
257
258 WIDECHAR* TempEnd = FPlatformString::Convert(Temp, ConvertedLen, Data, Len);
259 checkf(TempEnd!=nullptr, TEXT("String conversion unsuccessful"));
260
261 // This doesn't work for strings containing characters outside the BMP, but
262 // then neither does the WIDECHAR overload.
263 uint32 Result = FCrc::Strihash_DEPRECATED(ConvertedLen, Temp);
264
265 delete [] Temp;
266
267 return Result;
268}
