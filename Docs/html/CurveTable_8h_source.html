127 FRealCurve* FindCurve(FName RowName, const FString& ContextString, bool bWarnIfNotFound = true) const
128 {
129 if (RowName.IsNone())
130 {
131 UE_CLOG(bWarnIfNotFound, LogCurveTable, Warning, TEXT("UCurveTable::FindCurve : NAME_None is invalid row name for CurveTable '%s' (%s)."), *GetPathName(), *ContextString);
132 return nullptr;
133 }
134
135 FRealCurve* const* FoundCurve = RowMap.Find(RowName);
136
137 if (FoundCurve == nullptr)
138 {
139 UE_CLOG(bWarnIfNotFound, LogCurveTable, Warning, TEXT("UCurveTable::FindCurve : Row '%s' not found in CurveTable '%s' (%s)."), *RowName.ToString(), *GetPathName(), *ContextString);
140 return nullptr;
141 }
142
143 return *FoundCurve;
144 }

146 FRichCurve* FindRichCurve(FName RowName, const FString& ContextString, bool bWarnIfNotFound = true) const
147 {

150 UE_LOG(LogCurveTable, Error, TEXT("UCurveTable::FindCurve : Using FindRichCurve on CurveTable '%s' (%s) that is storing simple curves."), *GetPathName(), *ContextString);
151 return nullptr;

154 return (FRichCurve*)FindCurve(RowName, ContextString, bWarnIfNotFound);
