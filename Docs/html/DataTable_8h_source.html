231 template <class T>
232 T* FindRow(FName RowName, const TCHAR* ContextString, bool bWarnIfRowMissing = true) const
233 {
234 if(RowStruct == nullptr)
235 {
236 UE_LOG(LogDataTable, Error, TEXT("UDataTable::FindRow : '%s' specified no row for DataTable '%s'."), ContextString, *GetPathName());
237 return nullptr;
238 }
239
240 if(!RowStruct->IsChildOf(T::StaticStruct()))
241 {
242 UE_CLOG(bWarnIfRowMissing, LogDataTable, Error, TEXT("UDataTable::FindRow : '%s' specified incorrect type for DataTable '%s'."), ContextString, *GetPathName());
243 return nullptr;
244 }
245
246 if(RowName.IsNone())
247 {
248 UE_CLOG(bWarnIfRowMissing, LogDataTable, Warning, TEXT("UDataTable::FindRow : '%s' requested invalid row 'None' from DataTable '%s'."), ContextString, *GetPathName());
249 return nullptr;
250 }
251
252 uint8* const* RowDataPtr = GetRowMap().Find(RowName);
253 if (RowDataPtr == nullptr)
254 {
255 if (bWarnIfRowMissing)
256 {
257 UE_LOG(LogDataTable, Warning, TEXT("UDataTable::FindRow : '%s' requested row '%s' not in DataTable '%s'."), ContextString, *RowName.ToString(), *GetPathName());
258 }
259 return nullptr;
260 }
261
262 uint8* RowData = *RowDataPtr;
263 check(RowData);
264
265 return reinterpret_cast<T*>(RowData);
266 }

511struct FDataTableCategoryHandle
512{
513 GENERATED_USTRUCT_BODY()
514
515
516 UPROPERTY(EditAnywhere, BlueprintReadWrite, Category=DataTableCategoryHandle)
517 TObjectPtr<const class UDataTable> DataTable = nullptr;
518
520 UPROPERTY(EditAnywhere, BlueprintReadWrite, Category=DataTableCategoryHandle)
521 FName ColumnName;
522
524 UPROPERTY(EditAnywhere, BlueprintReadWrite, Category=DataTableCategoryHandle)
525 FName RowContents;
526
528 bool IsNull() const
529 {
530 return DataTable == nullptr && ColumnName == NAME_None && RowContents == NAME_None;
531 }
532
534 template <class T>
535 void GetRows(TArray<T*>& OutRows, const FString& ContextString) const
536 {
537 OutRows.Empty();
538 if (DataTable == nullptr)
539 {
540 if (RowContents != NAME_None)
541 {
542 UE_LOG(LogDataTable, Warning, TEXT("FDataTableCategoryHandle::GetRows : No DataTable for row %s (%s)."), *RowContents.ToString(), *ContextString);
543 }
544
545 return;
546 }
547
548 if (ColumnName == NAME_None)
549 {
550 if (RowContents != NAME_None)
551 {
552 UE_LOG(LogDataTable, Warning, TEXT("FDataTableCategoryHandle::GetRows : No Column selected for row %s (%s)."), *RowContents.ToString(), *ContextString);
553 }
554
555 return;
556 }
557
558 // Find the property that matches the desired column (ColumnName)
559 FProperty* Property = DataTable->FindTableProperty(ColumnName);
560 if (Property == nullptr)
561 {
562 return;
563 }
564
565 // check each row to see if the value in the Property element is the one we're looking for (RowContents). If it is, add the row to OutRows
566 uint8* RowContentsAsBinary = (uint8*)FMemory_Alloca(Property->GetSize());
567 Property->InitializeValue(RowContentsAsBinary);
568 if (Property->ImportText_Direct(*RowContents.ToString(), RowContentsAsBinary, nullptr, PPF_None) == nullptr)
569 {
570 Property->DestroyValue(RowContentsAsBinary);
571 return;
572 }
573
574 for (auto RowIt = DataTable->GetRowMap().CreateConstIterator(); RowIt; ++RowIt)
575 {
576 uint8* RowData = RowIt.Value();
577
578 if (Property->Identical(Property->ContainerPtrToValuePtr<void>(RowData, 0), RowContentsAsBinary, PPF_None))
579 {
580 OutRows.Add((T*)RowData);
581 }
582 }
583 Property->DestroyValue(RowContentsAsBinary);
584
585 return;
586 }

589 ENGINE_API bool operator!=(FDataTableCategoryHandle const& Other) const;
