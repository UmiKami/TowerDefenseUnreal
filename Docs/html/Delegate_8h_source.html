475 inline FName GetTrimmedMemberFunctionName(const TCHAR* InMacroFunctionName)
476 {
477 // We strip off the class prefix and just return the function name by itself.
478 check(InMacroFunctionName);
479 const TCHAR* Result = FCString::Strrstr( InMacroFunctionName, TEXT( "::" ) );
480 checkf(Result && Result[2] != (TCHAR)'0', TEXT("'%s' does not look like a member function"), InMacroFunctionName);
481 return FName(Result + 2);
482 }
483}
