32 // load half (F16) to float
33 //https://gist.github.com/rygorous/2156668
34 [[nodiscard]] static constexpr FORCEINLINE float LoadHalf(const uint16* Ptr)
35 {
36 uint16 FP16 = *Ptr;
37 constexpr uint32 shifted_exp = 0x7c00 << 13; // exponent mask after shift
38 union FP32T
39 {
40 uint32 u;
41 float f;
42 } FP32 = {}, magic = { 113 << 23 };
43
44 FP32.u = (FP16 & 0x7fff) << 13; // exponent/mantissa bits
45 uint32 exp = shifted_exp & FP32.u; // just the exponent
46 FP32.u += uint32(127 - 15) << 23; // exponent adjust
47
48 // handle exponent special cases
49 if (exp == shifted_exp) // Inf/NaN?
50 {
51 FP32.u += uint32(128 - 16) << 23; // extra exp adjust
52 }
53 else if (exp == 0) // Zero/Denormal?
54 {
55 FP32.u += 1 << 23; // extra exp adjust
56 FP32.f -= magic.f; // renormalize
57 }
58
59 FP32.u |= (FP16 & 0x8000) << 16; // sign bit
60 return FP32.f;
61 }

63 // store float to half (F16)
64 // converts with RTNE = round to nearest even
65 // values too large for F16 are stored as +-Inf
66 // https://gist.github.com/rygorous/2156668
67 // float_to_half_fast3_rtne
68 static constexpr FORCEINLINE void StoreHalf(uint16* Ptr, float Value)
69 {
70 union FP32T
71 {
72 uint32 u;
73 float f;
74 } FP32 = {};
75 uint16 FP16 = {};
76
77 FP32.f = Value;
78
79 constexpr FP32T f32infty = { uint32(255 << 23) };
80 constexpr FP32T f16max = { uint32(127 + 16) << 23 };
81 constexpr FP32T denorm_magic = { (uint32(127 - 15) + uint32(23 - 10) + 1) << 23 };
82 constexpr uint32 sign_mask = 0x80000000u;
83
84 uint32 sign = FP32.u & sign_mask;
85 FP32.u ^= sign;
86
87 // NOTE all the integer compares in this function can be safely
88 // compiled into signed compares since all operands are below
89 // 0x80000000. Important if you want fast straight SSE2 code
90 // (since there's no unsigned PCMPGTD).
91
92 if (FP32.u >= f16max.u) // result is Inf or NaN (all exponent bits set)
93 {
94 FP16 = (FP32.u > f32infty.u) ? 0x7e00 : 0x7c00; // NaN->qNaN and Inf->Inf
95 }
96 else // (De)normalized number or zero
97 {
98 if (FP32.u < uint32(113 << 23)) // resulting FP16 is subnormal or zero
99 {
100 // use a magic value to align our 10 mantissa bits at the bottom of
101 // the float. as long as FP addition is round-to-nearest-even this
102 // just works.
103 FP32.f += denorm_magic.f;
104
105 // and one integer subtract of the bias later, we have our final float!
106 FP16 = uint16(FP32.u - denorm_magic.u);
107 }
108 else
109 {
110 uint32 mant_odd = (FP32.u >> 13) & 1; // resulting mantissa is odd
111
112 // update exponent, rounding bias part 1
113 FP32.u += (uint32(15 - 127) << 23) + 0xfff;
114 // rounding bias part 2
115 FP32.u += mant_odd;
116 // take the bits!
117 FP16 = uint16(FP32.u >> 13);
118 }
119 }
120
121 FP16 |= sign >> 16;
122 *Ptr = FP16;
123 }

1185 typename IntType
1186 UE_REQUIRES(std::is_integral_v<IntType>)
1187 >
1188 [[nodiscard]] static FORCEINLINE bool AddAndCheckForOverflow(IntType A, IntType B, IntType& OutResult)
1189 {
1190 // This follows Hacker's Delight, Chapter 2-12
1191 // Signed->unsigned conversion and unsigned addition have defined behavior always
1192 typedef typename std::make_unsigned_t<IntType> UnsignedType;
1193 const UnsignedType UnsignedA = static_cast<UnsignedType>(A);
1194 const UnsignedType UnsignedB = static_cast<UnsignedType>(B);
1195 const UnsignedType UnsignedSum = UnsignedA + UnsignedB;
1196
1197 if constexpr (std::is_signed_v<IntType>)
1198 {
1199 // Check for signed overflow.
1200 // The underlying logic here is pretty simple: if A and B had opposite signs, their sum can't
1201 // overflow. If they had the same sign and the sum has the opposite value in the sign bit, we
1202 // had an overflow. (See Hacker's Delight Chapter 2-12 for more details.)
1203 constexpr UnsignedType UnsignedMSB = static_cast<UnsignedType>(std::numeric_limits<IntType>::min());
1204 if((UnsignedSum ^ UnsignedA) & (UnsignedSum ^ UnsignedB) & UnsignedMSB)
1205 {
1206 return false;
1207 }
1208 else
1209 {
1210 OutResult = A + B;
1211 return true;
1212 }
1213 }
1214 else
1215 {
1216 // Iff there was overflow, the modular sum must be less than both the operands.
1217 if (UnsignedSum >= UnsignedA)
1218 {
1219 OutResult = UnsignedSum;
1220 return true;
1221 }
1222 else
1223 {
1224 return false;
1225 }
1226 }

1238 [[nodiscard]] static FORCEINLINE bool SubtractAndCheckForOverflow(IntType A, IntType B, IntType& OutResult)
1239 {
1240 // This follows Hacker's Delight, Chapter 2-12
1241 // Signed->unsigned conversion and unsigned addition have defined behavior always
1242 typedef typename std::make_unsigned_t<IntType> UnsignedType;
1243 const UnsignedType UnsignedA = static_cast<UnsignedType>(A);

1252 // simultaneously the sign of the difference differs from the sign of the minuend (which should
1253 // keep its sign when we're subtracting a value of the opposite sign), meaning

1255 constexpr UnsignedType UnsignedMSB = static_cast<UnsignedType>(std::numeric_limits<IntType>::min());

1261 {

1278 template <
1279 typename IntType
1280 UE_REQUIRES(std::is_integral_v<IntType>)
1281 >
1282 [[nodiscard]] static FORCEINLINE bool MultiplyAndCheckForOverflow(IntType A, IntType B, IntType& OutResult)
1283 {
1284 // Handle the case where the second factor is 0 specially (why will become clear in a minute).
1285 if (B == 0)
1286 {
1287 // Anything times 0 is 0.
1288 OutResult = 0;
1289 return true;
1290 }
1291
1292 if constexpr (std::is_signed_v<IntType>)
1293 {
1294 // The overflow check is annoying and expensive, but the basic idea is fairly simple:
1295 // reduce to an unsigned check of the absolute values. (Again the basic algorithm is
1296 // in Hacker's Delight, Chapter 2-12).
1297 //
1298 // We need the absolute value of the product to be <=MaxValue when the result is positive
1299 // (signs of factors same) and <= -MinValue = MaxValue + 1 if the result is negative
1300 // (signs of factors opposite).
1301 typedef typename std::make_unsigned_t<IntType> UnsignedType;
1302 UnsignedType UnsignedA = static_cast<UnsignedType>(A);
1303 UnsignedType UnsignedB = static_cast<UnsignedType>(B);
1304 bool bSignsDiffer = false;
1305
1306 // Determine the unsigned absolute values of A and B carefully (note we can't negate signed
1307 // A or B, because negating MinValue is UB). We can however subtract their
1308 // unsigned values from 0 if the original value was less than zero. While doing this, also
1309 // keep track of the sign parity.
1310 if (A < 0)
1311 {
1312 UnsignedA = UnsignedType(0) - UnsignedA;
1313 bSignsDiffer = !bSignsDiffer;
1314 }
1315
1316 if (B < 0)
1317 {
1318 UnsignedB = UnsignedType(0) - UnsignedB;
1319 bSignsDiffer = !bSignsDiffer;
1320 }
1321
1322 // Determine the unsigned product bound we need based on whether the signs were same or different.
1323 const UnsignedType ProductBound = UnsignedType(std::numeric_limits<IntType>::max()) + (bSignsDiffer ? 1 : 0);
1324
1325 // We're now in the unsigned case, 0 <= UnsignedA, 0 < UnsignedB (we established b != 0), and for
1326 // there not to be overflows we need
1327 // a * b <= ProductBound
1328 // <=> a <= ProductBound/b
1329 // <=> a <= floor(ProductBound/b) since a is integer
1330 if (UnsignedA <= ProductBound / UnsignedB)
1331 {
1332 OutResult = A * B;
1333 return true;
1334 }
1335 else
1336 {
1337 return false;
1338 }
1339 }
1340 else
1341 {
1342 OutResult = A * B;
1343 return OutResult / B == A;
1344 }
1345 }
