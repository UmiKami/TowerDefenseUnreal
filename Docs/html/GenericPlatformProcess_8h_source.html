855 static FORCEINLINE void Yield()
856 {
857#if PLATFORM_USE_SSE2_FOR_THREAD_YIELD
858 _mm_pause();
859#elif PLATFORM_CPU_ARM_FAMILY
860# if !defined(__clang__)
861 __yield(); // MSVC
862# else
863 __builtin_arm_yield();
864# endif
865#else
866 // the platform with other architectures must override this to not have this function be called
867 unimplemented();
868#endif
869 }

875 static FORCEINLINE void YieldCycles(uint64 Cycles)
876 {
877#if PLATFORM_CPU_X86_FAMILY
878 auto ReadCycleCounter = []()
879 {
880#if defined(_MSC_VER)
881 return __rdtsc();
882#elif PLATFORM_APPLE
883 return mach_absolute_time();
884#elif __has_builtin(__builtin_readcyclecounter)
885 return __builtin_readcyclecounter();
886#else
887 // the platform with other architectures must override this to not have this function be called
888 unimplemented();
889#endif
890 };
891
892 uint64 start = ReadCycleCounter();
893 //some 32bit implementations return 0 for __builtin_readcyclecounter just to be on the safe side we protect against this.
894 Cycles = start != 0 ? Cycles : 0;
895
896#if PLATFORM_WINDOWS
897 if (FPlatformMisc::HasTimedPauseCPUFeature())
898 {
899 uint64 PauseCycles = ReadCycleCounter() + Cycles;
900#if defined(_MSC_VER)
901 _tpause(0, PauseCycles);
902#elif __has_builtin(__builtin_ia32_tpause)
903 __builtin_ia32_tpause(0, (uint32)(PauseCycles >> 32), (uint32)PauseCycles);
904#else
905# error Unsupported architecture!
906#endif
907 }
908 else
909#endif
910 {
911 do
912 {
913 Yield();
914 } while ((ReadCycleCounter() - start) < Cycles);
915 }
916
917#else
918 // We can't read cycle counter from user mode on these platform
919 for (uint64 i = 0; i < Cycles; i++)
920 {
921 Yield();
922 }
923#endif
924 }
