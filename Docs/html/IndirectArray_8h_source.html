12/*-----------------------------------------------------------------------------
13 Indirect array.

15 resizing the array index without relocating the actual elements.
16-----------------------------------------------------------------------------*/

487{
488 A.CountBytes(Ar);

501 }
502 else
503 {
504 // Save array.
505 int32 Num = A.Num();

507 for (int32 Index = 0; Index < Num; Index++)
508 {
509 Ar << A[Index];
510 }
511 }
512 return Ar;
