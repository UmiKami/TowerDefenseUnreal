142template<class TDataLayout>
143const TDataLayout* UInputDeviceProperty::GetDeviceSpecificData(const FPlatformUserId UserId, const FInputDeviceId DeviceId, const TMap<FName, TDataLayout>& InDeviceData) const
144{
145 if (const UInputDeviceSubsystem* SubSystem = UInputDeviceSubsystem::Get())
146 {
147 const FHardwareDeviceIdentifier Hardware = SubSystem->GetMostRecentlyUsedHardwareDevice(UserId);
148 // Check if there are any per-input device overrides available
149 if (const TDataLayout* DeviceDetails = InDeviceData.Find(Hardware.HardwareDeviceIdentifier))
150 {
151 return DeviceDetails;
152 }
153 }
154
155 return nullptr;
156}
