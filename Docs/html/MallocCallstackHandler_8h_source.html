177 struct FCallStackMapKey
178 {
179 uint32 CRC;
180 uint64* CallStack;
181
182 FCallStackMapKey(uint64* InCallStack)
183 : CallStack(InCallStack)
184 {
185 CRC = FCrc::MemCrc32(InCallStack, MaxCallStackDepth * sizeof(uint64));
186 }
187
188 friend bool operator==(const FCallStackMapKey& A, const FCallStackMapKey& B)
189 {
190 if (A.CRC != B.CRC)
191 {
192 return false;
193 }
194 for (int i = 0; i < MaxCallStackDepth; ++i)
195 {
196 uint64 APtr = A.CallStack[i];
197 uint64 BPtr = B.CallStack[i];
198 if (APtr != BPtr)
199 {
200 return false;
201 }
202 if (APtr == 0)
203 break;
204 }
205 return true;
206 }
207
208 friend inline uint32 GetTypeHash(const FCallStackMapKey& InKey)
209 {
210 return InKey.CRC;
211 }
212 };
