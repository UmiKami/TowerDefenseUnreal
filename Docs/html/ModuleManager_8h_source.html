981 { \
982 FSigningKeyRegistration() \
983 { \
984 extern CORE_API void RegisterSigningKeyCallback(void (*)(TArray<uint8>&, TArray<uint8>&)); \
985 RegisterSigningKeyCallback(&Callback); \
986 } \
987 static void Callback(TArray<uint8>& OutExponent, TArray<uint8>& OutModulus) \
988 { \
989 const uint8 Exponent[] = { ExponentValue }; \
990 const uint8 Modulus[] = { ModulusValue }; \
991 OutExponent.SetNum(UE_ARRAY_COUNT(Exponent)); \
992 OutModulus.SetNum(UE_ARRAY_COUNT(Modulus)); \
993 for(int ByteIdx = 0; ByteIdx < UE_ARRAY_COUNT(Exponent); ByteIdx++) \
994 { \
995 OutExponent[ByteIdx] = Exponent[ByteIdx]; \
996 } \
997 for(int ByteIdx = 0; ByteIdx < UE_ARRAY_COUNT(Modulus); ByteIdx++) \
998 { \
999 OutModulus[ByteIdx] = Modulus[ByteIdx]; \
