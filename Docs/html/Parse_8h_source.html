149 //
150 // Parse a hex digit.
151 //
152 static FORCEINLINE int32 HexDigit(TCHAR c)
153 {
154 int32 Result = 0;
155
156 if (c >= '0' && c <= '9')
157 {
158 Result = c - '0';
159 }
160 else if (c >= 'a' && c <= 'f')
161 {
162 Result = c + 10 - 'a';
163 }
164 else if (c >= 'A' && c <= 'F')
165 {
166 Result = c + 10 - 'A';
167 }
168 else
169 {
170 Result = 0;
171 }
172
173 return Result;
174 }
