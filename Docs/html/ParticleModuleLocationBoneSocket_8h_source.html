213 void InitArrayProxies(int32 FixedArraySize)
214 {
215 // Calculate offsets into instance data buffer for the arrays and initialize the buffer proxies. The allocation
216 // size for these arrays is calculated in RequiredBytesPerInstance.
217 const uint32 StructSize = sizeof(FModuleLocationBoneSocketInstancePayload);
218 PrevFrameBoneSocketPositions = TPreallocatedArrayProxy<FVector3f>((uint8*)this + StructSize, FixedArraySize);
219
220 uint32 StructOffset = StructSize + (FixedArraySize*sizeof(FVector3f));
221 BoneSocketVelocities = TPreallocatedArrayProxy<FVector3f>((uint8*)this + StructOffset, FixedArraySize);
222 StructOffset += (FixedArraySize*sizeof(FVector3f));
223
224 PreSelectedBoneSocketIndices = TPreallocatedArrayProxy<int32>((uint8*)this + StructOffset, FixedArraySize);
225 StructOffset += (FixedArraySize*sizeof(int32));
