40FORCEINLINE void AdjustParticleBaseSizeForUVFlipping(FVector& OutSize, EParticleUVFlipMode FlipMode, FRandomStream& InRandomStream)
41{
42 static const float HalfRandMax = 0.5f;
43
44 switch (FlipMode)
45 {
46 case EParticleUVFlipMode::None:
47 return;
48
49 case EParticleUVFlipMode::FlipUV:
50 OutSize = -OutSize;
51 return;
52
53 case EParticleUVFlipMode::FlipUOnly:
54 OutSize.X = -OutSize.X;
55 return;
56
57 case EParticleUVFlipMode::FlipVOnly:
58 OutSize.Y = -OutSize.Y;
59 return;
60
61 case EParticleUVFlipMode::RandomFlipUV:
62 OutSize = InRandomStream.FRand() > HalfRandMax ? -OutSize : OutSize;
63 return;
64
65 case EParticleUVFlipMode::RandomFlipUOnly:
66 OutSize.X = InRandomStream.FRand() > HalfRandMax ? -OutSize.X : OutSize.X;
67 return;
68
69 case EParticleUVFlipMode::RandomFlipVOnly:
70 OutSize.Y = InRandomStream.FRand() > HalfRandMax ? -OutSize.Y : OutSize.Y;
71 return;
72
73 case EParticleUVFlipMode::RandomFlipUVIndependent:
74 OutSize.X = InRandomStream.FRand() > HalfRandMax ? -OutSize.X : OutSize.X;
75 OutSize.Y = InRandomStream.FRand() > HalfRandMax ? -OutSize.Y : OutSize.Y;
76 return;
77
78 default:
79 checkNoEntry();
80 break;
81 }
82}
