178// Predicate to determine if an overlap is with a certain AActor.
179struct FPredicateOverlapHasSameActor
180{
181 FPredicateOverlapHasSameActor(const AActor& Owner)
182 : MyOwnerPtr(&Owner)
183 {
184 }
185
186 bool operator() (const FOverlapInfo& Info)
187 {
188 // MyOwnerPtr is always valid, so we don't need the IsValid() checks in the WeakObjectPtr comparison operator.
189 return MyOwnerPtr.HasSameIndexAndSerialNumber(Info.OverlapInfo.HitObjectHandle.FetchActor());
190 }
191
192private:
193 const TWeakObjectPtr<const AActor> MyOwnerPtr;
194};

196// Predicate to determine if an overlap is *NOT* with a certain AActor.
197struct FPredicateOverlapHasDifferentActor
198{
199 FPredicateOverlapHasDifferentActor(const AActor& Owner)
200 : MyOwnerPtr(&Owner)
201 {
202 }
203
204 bool operator() (const FOverlapInfo& Info)
205 {
206 // MyOwnerPtr is always valid, so we don't need the IsValid() checks in the WeakObjectPtr comparison operator.
207 return !MyOwnerPtr.HasSameIndexAndSerialNumber(Info.OverlapInfo.HitObjectHandle.FetchActor());
208 }
209
210private:
211 const TWeakObjectPtr<const AActor> MyOwnerPtr;

3265bool UPrimitiveComponent::ConvertSweptOverlapsToCurrentOverlaps(
3266 TArray<FOverlapInfo, AllocatorType>& OverlapsAtEndLocation, const TOverlapArrayView& SweptOverlaps, int32 SweptOverlapsIndex,
3267 const FVector& EndLocation, const FQuat& EndRotationQuat)
3268{

3270
3271 bool bResult = false;
