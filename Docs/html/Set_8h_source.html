320 void Empty(int32 ExpectedNumElements = 0)
321 {
322 // Empty the elements array, and reallocate it for the expected number of elements.
323 const int32 DesiredHashSize = Allocator::GetNumberOfHashBuckets(ExpectedNumElements);
324 const bool ShouldDoRehash = ShouldRehash(ExpectedNumElements, DesiredHashSize, EAllowShrinking::Yes);
325
326 if (!ShouldDoRehash)

329 UnhashElements();

333
334 // Resize the hash to the desired size for the expected number of elements.
335 if (ShouldDoRehash)
336 {
337 HashSize = DesiredHashSize;
338 Rehash();
339 }
340 }

384 {
385 // makes sense only when Number > Elements.Num() since TSparseArray::Reserve
386 // does any work only if that's the case

573 bool TryReplaceExisting(uint32 KeyHash, SetElementType& Element, SizeType& InOutElementIndex, bool* bIsAlreadyInSetPtr)
574 {
575 bool bIsAlreadyInSet = false;
576 if constexpr (!KeyFuncs::bAllowDuplicateKeys)
577 {
578 // If the set doesn't allow duplicate keys, check for an existing element with the same key as the element being added.
579
580 // Don't bother searching for a duplicate if this is the first element we're adding
581 if (Elements.Num() != 1)
582 {
583 SizeType ExistingIndex = FindIndexByHash(KeyHash, KeyFuncs::GetSetKey(Element.Value));
584 bIsAlreadyInSet = ExistingIndex != INDEX_NONE;
585 if (bIsAlreadyInSet)
586 {
587 // If there's an existing element with the same key as the new element, replace the existing element with the new element.
588 MoveByRelocate(Elements[ExistingIndex].Value, Element.Value);
589
590 // Then remove the new element.
591 Elements.RemoveAtUninitialized(InOutElementIndex);
592
593 // Then point the return value at the replaced element.
594 InOutElementIndex = ExistingIndex;
595 }
596 }
597 }
598 if (bIsAlreadyInSetPtr)
599 {
600 *bIsAlreadyInSetPtr = bIsAlreadyInSet;
601 }
602 return bIsAlreadyInSet;
603 }

608 if (!ConditionalRehash(Elements.Num(), EAllowShrinking::No))
609 {
610 // If the rehash didn't add the new element to the hash, add it.
611 LinkElement(ElementIndex, Element, KeyHash);
612 }

719 void RemoveByIndex(SizeType ElementIndex)
720 {
721 checkf(Elements.IsValidIndex(ElementIndex), TEXT("Invalid ElementIndex passed to TSet::RemoveByIndex"));
722
723 const SetElementType& ElementBeingRemoved = Elements[ElementIndex];
724
725 // Remove the element from the hash.
726 FSetElementId* HashPtr = Hash.GetAllocation();
727 FSetElementId* NextElementIndexIter = &HashPtr[ElementBeingRemoved.HashIndex];
728 for (;;)
729 {
730 SizeType NextElementIndex = NextElementIndexIter->AsInteger();
731 checkf(NextElementIndex != INDEX_NONE, TEXT("Corrupt hash"));
732
733 if (NextElementIndex == ElementIndex)
734 {
735 *NextElementIndexIter = ElementBeingRemoved.HashNextId;
736 break;
737 }
738
739 NextElementIndexIter = &Elements[NextElementIndex].HashNextId;
740 }
741
742 // Remove the element from the elements array.
743 Elements.RemoveAt(ElementIndex);
744 }

763 [[nodiscard]] SizeType FindIndexByHash(uint32 KeyHash, const ComparableKey& Key) const
764 {
765 if (Elements.Num() == 0)
766 {
767 return INDEX_NONE;
768 }
769
770 FSetElementId* HashPtr = Hash.GetAllocation();
771 SizeType ElementIndex = HashPtr[KeyHash & (HashSize - 1)].AsInteger();
772 for (;;)
773 {
774 if (ElementIndex == INDEX_NONE)
775 {
776 return INDEX_NONE;

780 {
781 // Return the first match, regardless of whether the set has multiple matches for the key or not.
782 return ElementIndex;

785 ElementIndex = Elements[ElementIndex].HashNextId.AsInteger();

887 template<typename ComparableKey>
888 FORCEINLINE int32 RemoveImpl(uint32 KeyHash, const ComparableKey& Key)
889 {
890 int32 NumRemovedElements = 0;
891
892 FSetElementId* NextElementId = &GetTypedHash(KeyHash);
893 while (NextElementId->IsValidId())
894 {
895 const int32 ElementIndex = NextElementId->AsInteger();
896 SetElementType& Element = Elements[ElementIndex];
897
898 if (KeyFuncs::Matches(KeyFuncs::GetSetKey(Element.Value), Key))
899 {
900 // This element matches the key, remove it from the set. Note that Remove sets *NextElementId to point to the next
901 // element after the removed element in the hash bucket.
902 RemoveByIndex(ElementIndex);
903 NumRemovedElements++;
904
905 if constexpr (!KeyFuncs::bAllowDuplicateKeys)
906 {
907 // If the hash disallows duplicate keys, we're done removing after the first matched key.
908 break;
909 }
910 }
911 else
912 {
913 NextElementId = &Element.HashNextId;
914 }
915 }
916
917 return NumRemovedElements;
918 }
