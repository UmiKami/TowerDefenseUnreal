131 void Copy(const FAnimationEvaluationContext& Other)
132 {
133 AnimInstance = Other.AnimInstance;
134 PostProcessAnimInstance = Other.PostProcessAnimInstance;
135 SkeletalMesh = Other.SkeletalMesh;
136 ComponentSpaceTransforms.Reset();
137 ComponentSpaceTransforms.Append(Other.ComponentSpaceTransforms);
138 BoneSpaceTransforms.Reset();
139 BoneSpaceTransforms.Append(Other.BoneSpaceTransforms);
140 CachedComponentSpaceTransforms.Reset();
141 CachedComponentSpaceTransforms.Append(Other.CachedComponentSpaceTransforms);
142 CachedBoneSpaceTransforms.Reset();
143 CachedBoneSpaceTransforms.Append(Other.CachedBoneSpaceTransforms);
144 RootBoneTranslation = Other.RootBoneTranslation;
145 Curve.CopyFrom(Other.Curve);
146 CachedCurve.CopyFrom(Other.CachedCurve);
147 bDoInterpolation = Other.bDoInterpolation;
148 bDoEvaluation = Other.bDoEvaluation;
149 bDuplicateToCacheBones = Other.bDuplicateToCacheBones;
150 bDuplicateToCacheCurve = Other.bDuplicateToCacheCurve;
151 bDuplicateToCachedAttributes = Other.bDuplicateToCachedAttributes;
152 bForceRefPose = Other.bForceRefPose;
153
154 CustomAttributes.CopyFrom(Other.CustomAttributes);
155 CachedCustomAttributes.CopyFrom(Other.CachedCustomAttributes);
156 }

161 PostProcessAnimInstance = nullptr;
162 SkeletalMesh = nullptr;
