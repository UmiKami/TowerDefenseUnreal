570 template <typename InitKeyType>
571 FORCEINLINE ElementType* AllocateMemoryForEmplace(InitKeyType&& InKey)
572 {
573 int32 InsertIndex = Algo::LowerBoundBy(Pairs, InKey, FKeyForward(), SortPredicate());
574 check(InsertIndex >= 0 && InsertIndex <= Pairs.Num());
575
576 ElementType* DataPtr = nullptr;
577 // Since we returned lower bound we already know InKey <= InsertIndex key. So if InKey is not < InsertIndex key, they must be equal
578 if (Pairs.IsValidIndex(InsertIndex) && !SortPredicate()(InKey, Pairs[InsertIndex].Key))
579 {
580 // Replacing element, delete old one
581 DataPtr = Pairs.GetData() + InsertIndex;
582 DestructItems(DataPtr, 1);
583 }
584 else
585 {
586 // Adding new one, this may reallocate Pairs
587 Pairs.InsertUninitialized(InsertIndex, 1);
588 DataPtr = Pairs.GetData() + InsertIndex;
589 }
590
591 return DataPtr;
592 }
