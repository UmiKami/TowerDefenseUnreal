371 inline FName ToName(ESoundAssetCompressionType InDecoderType)
372 {
373 switch (InDecoderType)
374 {
375 case ESoundAssetCompressionType::RADAudio: return NAME_RADA;
376 case ESoundAssetCompressionType::BinkAudio: return NAME_BINKA;
377 case ESoundAssetCompressionType::ADPCM: return NAME_ADPCM;
378 case ESoundAssetCompressionType::PCM: return NAME_PCM;
379 case ESoundAssetCompressionType::Opus: return NAME_OPUS;
380 case ESoundAssetCompressionType::PlatformSpecific: return NAME_PLATFORM_SPECIFIC;
381 case ESoundAssetCompressionType::ProjectDefined: return NAME_PROJECT_DEFINED;
382 default:
383 ensure(false);
384 return TEXT("UNKNOWN");
385 }
386 }

395 case EDefaultAudioCompressionType::PCM: return ESoundAssetCompressionType::PCM;

397 case EDefaultAudioCompressionType::PlatformSpecific: return ESoundAssetCompressionType::PlatformSpecific;
