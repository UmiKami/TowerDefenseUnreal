66 // returns empty TOptional if queue is empty
67 TOptional<ElementType> Dequeue()
68 {
69 FNode* LocalTail = Tail.load(std::memory_order_relaxed);
70 FNode* LocalTailNext = LocalTail->Next.load(std::memory_order_acquire);
71 if (LocalTailNext == nullptr)
72 {
73 return {};
74 }
75
76 ElementType* TailNextValue = (ElementType*)&LocalTailNext->Value;
77 TOptional<ElementType> Value{ MoveTemp(*TailNextValue) };
78 DestructItem(TailNextValue);
79
80 Tail.store(LocalTailNext, std::memory_order_release);
81 return Value;
82 }

84 bool Dequeue(ElementType& OutElem)
