745 static FStatMessage ComputeCall(FStatMessage const& ScopeStart, FStatMessage const& ScopeEnd)
746 {
747 checkStats(ScopeStart.NameAndInfo.GetField<EStatOperation>() == EStatOperation::CycleScopeStart);
748 checkStats(ScopeEnd.NameAndInfo.GetField<EStatOperation>() == EStatOperation::CycleScopeEnd);
749 FStatMessage Result(ScopeStart);
750 Result.NameAndInfo.SetField<EStatOperation>(EStatOperation::Set);
751 Result.NameAndInfo.SetFlag(EStatMetaFlags::IsPackedCCAndDuration, true);
752 checkStats(ScopeEnd.NameAndInfo.GetFlag(EStatMetaFlags::IsCycle));
753
754 // cycles can wrap and are actually uint32's so we do something special here
755 int64 Delta = int32(uint32(ScopeEnd.GetValue_int64()) - uint32(ScopeStart.GetValue_int64()));
756 checkStats(Delta >= 0);
757 Result.GetValue_int64() = ToPackedCallCountDuration(1, uint32(Delta));
758 return Result;
759 }

802 // Extract the thread id.

804 int32 Index = 0;
