935template <typename CallableT>
936void FStreamableHandle::ForEachLoadedAsset(CallableT Callable) const
937{
938 if (HasLoadCompleted())
939 {
940 for (const FSoftObjectPath& Ref : RequestedAssets)
941 {
942 // Try manager, should be faster and will handle redirects better
943 if (IsActive())
944 {
945 Invoke(Callable, OwningManager->GetStreamed(Ref));
946 }
947 else
948 {
949 Invoke(Callable, Ref.ResolveObject());
950 }
951 }
952
953 // Check child handles
954 for (const TSharedPtr<FStreamableHandle>& ChildHandle : ChildHandles)
955 {
956 for (const FSoftObjectPath& Ref : ChildHandle->RequestedAssets)
957 {
958 // Try manager, should be faster and will handle redirects better
959 if (IsActive())
960 {
961 Invoke(Callable, OwningManager->GetStreamed(Ref));
962 }
963 else
964 {
965 Invoke(Callable, Ref.ResolveObject());
966 }
967 }
968 }
969 }

972template< typename PathContainerType, typename FuncType, typename >
973TSharedPtr<FStreamableHandle> FStreamableManager::RequestAsyncLoad(
974 PathContainerType&& TargetsToStream,
975 FuncType&& Callback,
976 TAsyncLoadPriority Priority,
977 bool bManageActiveHandle,
978 bool bStartStalled,
979 FString DebugName,
980 UE::FSourceLocation Location)
981{
982 FStreamableAsyncLoadParams Params;
983 Params.TargetsToStream = TArray<FSoftObjectPath>{ Forward<PathContainerType>(TargetsToStream) };
984
985 if constexpr (std::is_constructible_v<FStreamableDelegateWithHandle, FuncType>)
986 {
987 Params.OnComplete = FStreamableDelegateWithHandle(Forward<FuncType>(Callback));
988 }
989 else if constexpr (std::is_constructible_v<FStreamableDelegate, FuncType>)
990 {
991 Params.OnComplete = UE::StreamableManager::Private::WrapDelegate(
992 FStreamableDelegate(Forward<FuncType>(Callback)));
993 }
994 else if constexpr (std::is_invocable_v<FuncType, TSharedPtr<FStreamableHandle>>)
995 {
996 Params.OnComplete = FStreamableDelegateWithHandle::CreateLambda(Forward<FuncType>(Callback));
997 }
998 else if constexpr (std::is_invocable_v<FuncType>)
