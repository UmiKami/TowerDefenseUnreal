241 static int32 LoadStringTableAsset(const FName InTableId, FLoadStringTableAssetCallback InLoadedCallback = FLoadStringTableAssetCallback())

245 if (InstancePtr)

247 return InstancePtr->LoadStringTableAssetImpl(InTableId, InLoadedCallback);
