313template <typename ValueType UE_REQUIRES(TModels_V<UE::CSerializableForLog, ValueType>)>
314inline void SerializeForLog(FCbWriter& Writer, const TOptional<ValueType>& Optional)
315{
316 Writer.BeginArray();
317 if (Optional)
318 {
319 UE::CallSerializeForLog(Writer, Optional.GetValue());

321 Writer.EndArray();

438 template <typename... FieldArgTypes>

464template <typename LogType, typename... FieldArgTypes>
465UE_COLD UE_DEBUG_SECTION void LogWithFields(const FLogCategoryBase& Category, const LogType& Log, FieldArgTypes&&... FieldArgs)
466{
467 if constexpr (sizeof...(FieldArgTypes) == 0)
468 {
469 LogWithNoFields(Category, Log);
470 }
471 else
472 {
473 constexpr int32 FieldCount = FLogFieldCreator::template GetCount<FieldArgTypes...>();
474 static_assert(FieldCount > 0);
475 FLogField Fields[FieldCount];
476 FLogFieldCreator::Create(Fields, (FieldArgTypes&&)FieldArgs...);
477 LogWithFieldArray(Category, Log, Fields, FieldCount);
478 }
479}
480
482template <typename LogType, typename... FieldArgTypes>
483[[noreturn]] UE_COLD UE_DEBUG_SECTION void FatalLogWithFields(const FLogCategoryBase& Category, const LogType& Log, FieldArgTypes&&... FieldArgs)
484{
485 if constexpr (sizeof...(FieldArgTypes) == 0)
486 {
487 FatalLogWithNoFields(Category, Log);
488 }
489 else
490 {
491 constexpr int32 FieldCount = FLogFieldCreator::template GetCount<FieldArgTypes...>();
492 static_assert(FieldCount > 0);
493 FLogField Fields[FieldCount];
494 FLogFieldCreator::Create(Fields, (FieldArgTypes&&)FieldArgs...);
495 FatalLogWithFieldArray(Category, Log, Fields, FieldCount);
