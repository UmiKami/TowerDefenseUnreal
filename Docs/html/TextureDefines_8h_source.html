515FORCEINLINE bool IsHDR(ETextureSourceFormat Format)
516{
517 return (Format == TSF_BGRE8 || Format == TSF_RGBA16F || Format == TSF_RGBA32F || Format == TSF_R16F || Format == TSF_R32F);
518}
519
520FORCEINLINE bool IsHDR(TextureCompressionSettings CompressionSettings)
521{
522 switch(CompressionSettings)
523 {
524 case TC_HDR:
525 case TC_HDR_F32:
526 case TC_HDR_Compressed:
527 case TC_HalfFloat:
528 case TC_SingleFloat:
529 return true;
530 default:
531 return false;
532 }
533}
534
535FORCEINLINE bool IsUncompressed(TextureCompressionSettings CompressionSettings)
536{
537 return (CompressionSettings == TC_Grayscale ||
538 CompressionSettings == TC_Displacementmap ||
539 CompressionSettings == TC_VectorDisplacementmap ||
540 CompressionSettings == TC_HDR ||
541 CompressionSettings == TC_HDR_F32 ||
542 CompressionSettings == TC_EditorIcon ||
543 CompressionSettings == TC_DistanceFieldFont ||
544 CompressionSettings == TC_HalfFloat ||
545 CompressionSettings == TC_SingleFloat
546 );
547}
548
549FORCEINLINE bool ShouldUseGreyScaleEditorVisualization(TextureCompressionSettings CompressionSettings)
550{
551 // these formats should do R -> RGB red to gray replication in Editor viz
552 return (CompressionSettings == TC_Grayscale ||
553 CompressionSettings == TC_Alpha ||
554 CompressionSettings == TC_Displacementmap ||
555 CompressionSettings == TC_DistanceFieldFont );
556
557 // ?? maybe these too ??
558 // CompressionSettings == TC_HalfFloat ||
559 // CompressionSettings == TC_SingleFloat
560}
