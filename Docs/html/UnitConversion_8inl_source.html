75template<typename T>
76T FUnitConversion::Convert(T InValue, EUnit From, EUnit To)
77{
78 using namespace UnitConversion;
79
80 if (!AreUnitsCompatible(From, To))
81 {
82 return InValue;
83 }
84 else if (From == EUnit::Unspecified || To == EUnit::Unspecified)
85 {
86 return InValue;
87 }
88
89 switch(FUnitConversion::GetUnitType(From))
90 {
91 case EUnitType::Distance: return InValue * DistanceUnificationFactor(From) * (1.0 / DistanceUnificationFactor(To));
92 case EUnitType::Angle: return InValue * AngleUnificationFactor(From) * (1.0 / AngleUnificationFactor(To));
93 case EUnitType::Speed: return InValue * SpeedUnificationFactor(From) * (1.0 / SpeedUnificationFactor(To));
94 case EUnitType::AngularSpeed: return InValue * AngularSpeedUnificationFactor(From) * (1.0 / AngularSpeedUnificationFactor(To));
95 case EUnitType::Acceleration: return InValue * AccelerationUnificationFactor(From) * (1.0 / AccelerationUnificationFactor(To));
96 case EUnitType::Mass: return InValue * MassUnificationFactor(From) * (1.0 / MassUnificationFactor(To));
97 case EUnitType::Density: return InValue * DensityUnificationFactor(From) * (1.0 / DensityUnificationFactor(To));
98 case EUnitType::Force: return InValue * ForceUnificationFactor(From) * (1.0 / ForceUnificationFactor(To));
99 case EUnitType::Torque: return InValue * TorqueUnificationFactor(From) * (1.0 / TorqueUnificationFactor(To));
100 case EUnitType::PositionalImpulse: return InValue * PositionalImpulseUnificationFactor(From) * (1.0 / PositionalImpulseUnificationFactor(To));
101 case EUnitType::Frequency: return InValue * FrequencyUnificationFactor(From) * (1.0 / FrequencyUnificationFactor(To));
102 case EUnitType::DataSize: return InValue * DataSizeUnificationFactor(From) * (1.0 / DataSizeUnificationFactor(To));
103 case EUnitType::LuminousFlux: return InValue;
104 case EUnitType::Time: return InValue * TimeUnificationFactor(From) * (1.0 / TimeUnificationFactor(To));
105 case EUnitType::Multipliers: return InValue * MultiplierUnificationFactor(From) * (1.0 / MultiplierUnificationFactor(To));
106 case EUnitType::Stress: return InValue * StressUnificationFactor(From) * (1.0 / StressUnificationFactor(To));
107 // Temperature conversion is not just a simple multiplication, so needs special treatment
108 case EUnitType::Temperature:
109 {
110 double NewValue = InValue;
111 // Put it into kelvin
112 if (From == EUnit::Celsius)
113 {
114 NewValue = NewValue + 273.15f;
115 }
116 else if (From == EUnit::Farenheit)
117 {
118 NewValue = (NewValue + 459.67f) * 5.f/9.f;
119 }
120 // And out again
121 if (To == EUnit::Celsius)
122 {
123 return NewValue - 273.15f;
124 }
125 else if (To == EUnit::Farenheit)
126 {
127 return NewValue * 9.f/5.f - 459.67f;
128 }
129 else
130 {
131 return NewValue;
132 }
133 }
134
135 case EUnitType::Impulse:
136 case EUnitType::LuminousIntensity:
137 case EUnitType::Illuminance:
138 case EUnitType::Luminance:
139 case EUnitType::PixelDensity:
140 case EUnitType::ExposureValue:
141 case EUnitType::NumberOf:
142 default: return InValue;
143 }
144}

146template<typename T>
147FNumericUnit<T> FUnitConversion::QuantizeUnitsToBestFit(T Value, EUnit Units)
148{
149 auto OptionalBounds = UnitConversion::GetQuantizationBounds(Units);
150 if (!OptionalBounds.IsSet())
151 {
152 return FNumericUnit<T>(Value, Units);
153 }
154
155 const auto& Bounds = *OptionalBounds.GetValue();
156
157 const int32 CurrentUnitIndex = (uint8)Units - (uint8)Bounds[0].Units;
158
159 EUnit NewUnits = Units;
160 double NewValue = Value;
161
162 if (FMath::Abs(NewValue) > 1)
163 {
164 // Large number? Try larger units
165 for (int32 Index = CurrentUnitIndex; Index < Bounds.Num(); ++Index)
166 {
167 if (Bounds[Index].Factor == 0)
168 {
169 break;
170 }
171
172 const auto Tmp = NewValue / Bounds[Index].Factor;
173
174 if (FMath::Abs(Tmp) < 1)
175 {
176 break;
177 }
178
179 NewValue = Tmp;
180 NewUnits = Bounds[Index + 1].Units;
181 }
182 }
183 else if (NewValue != 0)
184 {
185 // Small number? Try smaller units
186 for (int32 Index = CurrentUnitIndex - 1; Index >= 0; --Index)
187 {
188 NewValue *= Bounds[Index].Factor;
189 NewUnits = Bounds[Index].Units;
190
191 if (FMath::Abs(NewValue) > 1)
192 {
193 break;
194 }
195 }
196 }
197
198 return FNumericUnit<T>(NewValue, NewUnits);
199}

201template<typename T>
202EUnit FUnitConversion::CalculateDisplayUnit(T Value, EUnit InUnits)
203{
204 if (InUnits == EUnit::Unspecified)
205 {
206 return EUnit::Unspecified;
207 }
208
209 const TArray<EUnit>& DisplayUnits = Settings().GetDisplayUnits(GetUnitType(InUnits));
210 if (DisplayUnits.Num() == 0)
211 {
212 return QuantizeUnitsToBestFit(Value, InUnits).Units;
213 }
214 if (DisplayUnits.Num() == 1)
215 {
216 return DisplayUnits[0];
217 }
218
219 // If the value we were given was 0, change it to something we can actually work with
220 if (Value == 0)
221 {
222 Value = 1;
223 }
224
225 int32 BestIndex = 0;
226 for (int32 Index = 0; Index < DisplayUnits.Num() - 1; ++Index)
227 {
228 T Best, Next;
229 if constexpr (std::is_signed_v<T>)
230 {
231 Best = FMath::Abs(Convert(Value, InUnits, DisplayUnits[BestIndex]));
232 Next = FMath::Abs(Convert(Value, InUnits, DisplayUnits[Index + 1]));
233 }
234 else
235 {
236 Best = Convert(Value, InUnits, DisplayUnits[BestIndex]);
237 Next = Convert(Value, InUnits, DisplayUnits[Index + 1]);
238 }
239
240 if (Best < 1.0 && Next >= 1.0)
241 {
242 BestIndex = Index + 1;
243 }
244 else if (Best < 1.0 && Next < 1.0)
245 {
246 if (Next > Best)
247 {
248 BestIndex = Index + 1;
249 }
250 }
251 else if (Best >= 1.0 && Next >= 1.0)
252 {
253 if (FMath::LogX(10.0f, Next) < FMath::LogX(10.0f, Best))
254 {
255 BestIndex = Index + 1;
256 }
257 }
258 }
259
260 return DisplayUnits[BestIndex];
261}

337template<typename NumericType>
338TOptional<FNumericUnit<NumericType>> FNumericUnit<NumericType>::TryParseString(const TCHAR* InSource)
339{
340 TOptional<FNumericUnit<NumericType>> Result;
341 if (!InSource || !*InSource)
342 {
343 return Result;
344 }
345
346 const TCHAR* NumberEnd = nullptr;
347 if (!ExtractNumberBoundary(InSource, NumberEnd))
348 {
349 return Result;
350 }
351
352 NumericType NewValue;
353 LexFromString(NewValue, InSource);
354
355 // Now parse the units
356 while(FChar::IsWhitespace(*NumberEnd)) ++NumberEnd;
357
358 if (*NumberEnd == '\0')
359 {
360 // No units
361 Result.Emplace(NewValue);
362 }
363 else
364 {
365 // If the string specifies units, they must map to something that exists for this function to succeed
366 auto NewUnits = FUnitConversion::UnitFromString(NumberEnd);
367 if (NewUnits)
368 {
369 Result.Emplace(NewValue, NewUnits.GetValue());
370 }
371 }
372
373 return Result;
374}

377template<typename NumericType> template<typename OtherType>
378void FNumericUnit<NumericType>::CopyValueWithConversion(const FNumericUnit<OtherType>& Other)
379{
380 if (Units != EUnit::Unspecified && Other.Units != EUnit::Unspecified)
381 {
382 if (Units == Other.Units)
383 {
384 Value = Other.Value;
385 }
386 else if (FUnitConversion::AreUnitsCompatible(Units, Other.Units))
387 {
388 Value = FUnitConversion::Convert(Other.Value, Other.Units, Units);
389 }
390 else
391 {
392 // Invalid conversion - assignment invalid
393 }
394 }
395 else
396 {
397 // If our units haven't been specified, we take on the units of the rhs
398 if (Units == EUnit::Unspecified)
399 {
400 // This is the only time we ever change units. Const_cast is 'acceptible' here since the units haven't been specified yet.
401 const_cast<EUnit&>(Units) = Other.Units;
402 }
403
404 Value = Other.Value;
405 }

408template<typename NumericType>
409bool FNumericUnit<NumericType>::ExtractNumberBoundary(const TCHAR* Start, const TCHAR*& End)
410{
411 while(FChar::IsWhitespace(*Start)) ++Start;
412
413 End = Start;
414 if (*End == '-' || *End == '+')
415 {
416 End++;
417 }
418
419 bool bHasDot = false;
420 while (FChar::IsDigit(*End) || *End == '.')
421 {
422 if (*End == '.')
423 {
424 if (bHasDot)
425 {
426 return false;
427 }
428 bHasDot = true;
429 }
430 ++End;
431 }
432
433 return true;
434}
