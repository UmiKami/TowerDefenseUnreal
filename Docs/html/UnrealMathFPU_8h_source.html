1443 const float T4 = (A[2] - A[0]) * (B[0] - B[1]);

1448 const float T9 = 0.5f * (T4 + T8);

1451 R[1] = T2 + T9 - T7;
1452 R[2] = T3 + T9 - T6;
1453 R[3] = T0 + T9 - T5;
1454#else
