423// Returns ((Vector dot Vector) >= 1e-8) ? (Vector / |Vector|) : DefaultValue
424// Uses accurate 1/sqrt, not the estimate
425FORCEINLINE VectorRegister4Float VectorNormalizeSafe(VectorRegister4Float Vector, VectorRegister4Float DefaultValue)
426{
427 const VectorRegister4Float SquareSum = VectorDot4(Vector, Vector);
428 const VectorRegister4Float NonZeroMask = VectorCompareGE(SquareSum, GlobalVectorConstants::SmallLengthThreshold);
429 const VectorRegister4Float InvLength = VectorReciprocalSqrtAccurate(SquareSum);
430 const VectorRegister4Float NormalizedVector = VectorMultiply(InvLength, Vector);
431 return VectorSelect(NonZeroMask, NormalizedVector, DefaultValue);
432}
433
434FORCEINLINE VectorRegister4Double VectorNormalizeSafe(VectorRegister4Double Vector, VectorRegister4Double DefaultValue)
435{
436 const VectorRegister4Double SquareSum = VectorDot4(Vector, Vector);
437 const VectorRegister4Double NonZeroMask = VectorCompareGE(SquareSum, GlobalVectorConstants::DoubleSmallLengthThreshold);
438 const VectorRegister4Double InvLength = VectorReciprocalSqrtAccurate(SquareSum);
439 const VectorRegister4Double NormalizedVector = VectorMultiply(InvLength, Vector);
440 return VectorSelect(NonZeroMask, NormalizedVector, DefaultValue);
441}

570// A and B are quaternions. The result is A + (|A.B| >= 0 ? 1 : -1) * B
571FORCEINLINE VectorRegister4Float VectorAccumulateQuaternionShortestPath(VectorRegister4Float A, VectorRegister4Float B)
572{
573 // Blend rotation
574 // To ensure the 'shortest route', we make sure the dot product between the both rotations is positive.
575 // const float Bias = (|A.B| >= 0 ? 1 : -1)
576 // return A + B * Bias;
577 const VectorRegister4Float Zero = VectorZeroFloat();
578 const VectorRegister4Float RotationDot = VectorDot4(A, B);
579 const VectorRegister4Float QuatRotationDirMask = VectorCompareGE(RotationDot, Zero);
580 const VectorRegister4Float NegativeB = VectorSubtract(Zero, B);
581 const VectorRegister4Float BiasTimesB = VectorSelect(QuatRotationDirMask, B, NegativeB);
582 return VectorAdd(A, BiasTimesB);
583}

610// VectorMod360: Essentially VectorMod(X, 360) but using faster computation that is still accurate given the known input constraints.
611FORCEINLINE VectorRegister4Float VectorMod360(VectorRegister4Float X)
612{
613 // R = X - (Y * (Trunc(X / Y))
614 VectorRegister4Float Temp = VectorTruncate(VectorDivide(X, GlobalVectorConstants::Float360));
615 VectorRegister4Float FloatResult = VectorNegateMultiplyAdd(GlobalVectorConstants::Float360, Temp, X);
616 return FloatResult;
617}

627// Normalize Rotator
628FORCEINLINE VectorRegister4Float VectorNormalizeRotator(VectorRegister4Float UnnormalizedRotator)
629{
630 // shift in the range [-360,360]
631 VectorRegister4Float V0 = VectorMod360(UnnormalizedRotator);
632 VectorRegister4Float V1 = VectorAdd( V0, GlobalVectorConstants::Float360 );
633 VectorRegister4Float V2 = VectorSelect(VectorCompareGE(V0, VectorZeroFloat()), V0, V1);
634
635 // shift to [-180,180]
636 VectorRegister4Float V3 = VectorSubtract( V2, GlobalVectorConstants::Float360 );
637 VectorRegister4Float V4 = VectorSelect(VectorCompareGT(V2, GlobalVectorConstants::Float180), V3, V2);
638
639 return V4;
640}

660FORCEINLINE VectorRegister4Float VectorLerpQuat(VectorRegister4Float A, VectorRegister4Float B, VectorRegister4Float Alpha)
661{
662 // Blend rotation
663 // To ensure the 'shortest route', we make sure the dot product between the both rotations is positive.
664 // const float Bias = (|A.B| >= 0 ? 1 : -1)
665 // Rotation = (B * Alpha) + (A * (Bias * (1.f - Alpha)));
666 const VectorRegister4Float Zero = VectorZeroFloat();
667
668 const VectorRegister4Float OneMinusAlpha = VectorSubtract(VectorOneFloat(), Alpha);
669
670 const VectorRegister4Float RotationDot = VectorDot4(A, B);
671 const VectorRegister4Float QuatRotationDirMask = VectorCompareGE(RotationDot, Zero);
672 const VectorRegister4Float NegativeA = VectorSubtract(Zero, A);
673 const VectorRegister4Float BiasTimesA = VectorSelect(QuatRotationDirMask, A, NegativeA);
674 const VectorRegister4Float BTimesWeight = VectorMultiply(B, Alpha);
675 const VectorRegister4Float UnnormalizedResult = VectorMultiplyAdd(BiasTimesA, OneMinusAlpha, BTimesWeight);
676
677 return UnnormalizedResult;
678}

741FORCEINLINE VectorRegister4Float VectorQuaternionRotateVector(VectorRegister4Float Quat, VectorRegister4Float VectorW0)
742{
743 // Q * V * Q.Inverse
744 //const VectorRegister InverseRotation = VectorQuaternionInverse(Quat);
745 //const VectorRegister Temp = VectorQuaternionMultiply2(Quat, VectorW0);
746 //const VectorRegister Rotated = VectorQuaternionMultiply2(Temp, InverseRotation);
747
748 // Equivalence of above can be shown to be:
749 // http://people.csail.mit.edu/bkph/articles/Quaternions.pdf
750 // V' = V + 2w(Q x V) + (2Q x (Q x V))
751 // refactor:
752 // V' = V + w(2(Q x V)) + (Q x (2(Q x V)))
753 // T = 2(Q x V);
754 // V' = V + w*(T) + (Q x T)
755
756 const VectorRegister4Float QW = VectorReplicate(Quat, 3);
757 VectorRegister4Float T = VectorCross(Quat, VectorW0);
758 T = VectorAdd(T, T);
759 const VectorRegister4Float VTemp0 = VectorMultiplyAdd(QW, T, VectorW0);
760 const VectorRegister4Float VTemp1 = VectorCross(Quat, T);
761 const VectorRegister4Float Rotated = VectorAdd(VTemp0, VTemp1);
762 return Rotated;
763}

765FORCEINLINE VectorRegister4Double VectorQuaternionRotateVector(VectorRegister4Double Quat, VectorRegister4Double VectorW0)
766{
767 const VectorRegister4Double QW = VectorReplicate(Quat, 3);
768 VectorRegister4Double T = VectorCross(Quat, VectorW0);
769 T = VectorAdd(T, T);
770 const VectorRegister4Double VTemp0 = VectorMultiplyAdd(QW, T, VectorW0);
771 const VectorRegister4Double VTemp1 = VectorCross(Quat, T);
772 const VectorRegister4Double Rotated = VectorAdd(VTemp0, VTemp1);
773 return Rotated;
