177 [[nodiscard]] FORCEINLINE UE_STRING_CLASS(const CFStringRef In)
178 {
179 uint32_t StringLength = In ? CFStringGetLength(In) : 0;
180
181 if (StringLength > 0)
182 {
183 // Convert the NSString data into the native TCHAR format for UE4
184 // This returns a buffer of bytes, but they can be safely cast to a buffer of TCHARs
185#if PLATFORM_TCHAR_IS_4_BYTES
186 const CFStringEncoding Encoding = kCFStringEncodingUTF32LE;
187#else
188 const CFStringEncoding Encoding = kCFStringEncodingUTF16LE;
189#endif
190
191 CFRange Range = CFRangeMake(0, StringLength);
192 CFIndex BytesNeeded;
193 if (CFStringGetBytes(In, Range, Encoding, '?', false, NULL, 0, &BytesNeeded) > 0)
194 {
195 const size_t Length = BytesNeeded / sizeof(TCHAR);
196 Data.Reserve(Length + 1);
197 Data.AddUninitialized(Length + 1);
198 CFStringGetBytes(In, Range, Encoding, '?', false, (uint8*)Data.GetData(), Length * sizeof(TCHAR) + 1, NULL);
199 Data[Length] = 0;
200 }
201 }
202 }
