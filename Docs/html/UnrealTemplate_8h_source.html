150template<typename OutType, typename InType>
151constexpr bool IntFitsIn(InType In)
152{
153 static_assert(std::is_integral_v<InType> && std::is_integral_v<OutType>, "Only integers supported");
154
155 OutType Out = static_cast<OutType>(In);
156 bool bRoundtrips = In == static_cast<InType>(Out);
157
158 // Signed <-> unsigned cast requires sign test, signed -> smaller signed is covered by roundtrip sign-extension.
159 if constexpr ((static_cast<InType>(-1) < InType{}) != (static_cast<OutType>(-1) < OutType{}))
160 {
161 return bRoundtrips && (In < InType{} == Out < OutType{});
162 }
163 else
164 {
165 return bRoundtrips;
166 }
167}

170template<typename OutType, typename InType>
171OutType IntCastChecked(InType In)
172{
173 if constexpr (std::is_signed_v<InType>)
174 {
175 checkf(IntFitsIn<OutType>(In), TEXT("Loss of data caused by narrowing conversion, In = %" INT64_FMT), (int64)In);
176 }
177 else
178 {
179 checkf(IntFitsIn<OutType>(In), TEXT("Loss of data caused by narrowing conversion, In = %" UINT64_FMT), (uint64)In);
180 }
181 return static_cast<OutType>(In);
182}

185template<typename OutType, typename InType>
186constexpr bool FloatFitsIn(InType In, InType Precision)
187{
188 static_assert(std::is_floating_point_v<InType> && std::is_floating_point_v<OutType>, "Only floating point supported");
189
190 OutType Out = static_cast<OutType>(In);
191 return fabs(static_cast<InType>(Out) - In) <= Precision;
192}

194template<typename OutType, typename InType>
195OutType FloatCastChecked(InType In, InType Precision)
196{
197 checkf(FloatFitsIn<OutType>(In, Precision), TEXT("Loss of data caused by narrowing conversion"));
198 return static_cast<OutType>(In);
199}

627template <typename T>
628constexpr inline void Swap(T& A, T& B)
629{
630 // std::is_swappable isn't correct here, because we allow bitwise swapping of types containing e.g. const and reference members,
631 // but we don't want to allow swapping of types which are UE_NONCOPYABLE or equivalent. We also allow bitwise swapping of arrays, so
632 // extents should be removed first.
633 static_assert(std::is_move_constructible_v<std::remove_all_extents_t<T>>, "Cannot swap non-movable types");
634
635 if constexpr (TUseBitwiseSwap<T>::Value)
636 {
637 struct FAlignedBytes
638 {
639 alignas(T) char Bytes[sizeof(T)];
640 };
641
642 FAlignedBytes Temp;
643 *(FAlignedBytes*)&Temp = *(FAlignedBytes*)&A;
644 *(FAlignedBytes*)&A = *(FAlignedBytes*)&B;
645 *(FAlignedBytes*)&B = *(FAlignedBytes*)&Temp;
646 }
647 else
648 {
649 T Temp = MoveTemp(A);
650 A = MoveTemp(B);
651 B = MoveTemp(Temp);
652 }
653}

656constexpr void Exchange(T& A, T& B)

685template <
686 typename T
687 UE_REQUIRES(std::is_same_v<T, uint32>)
688>
689FORCEINLINE T ReverseBits( T Bits )
690{
691 Bits = ( Bits << 16) | ( Bits >> 16);
692 Bits = ( (Bits & 0x00ff00ff) << 8 ) | ( (Bits & 0xff00ff00) >> 8 );
693 Bits = ( (Bits & 0x0f0f0f0f) << 4 ) | ( (Bits & 0xf0f0f0f0) >> 4 );
694 Bits = ( (Bits & 0x33333333) << 2 ) | ( (Bits & 0xcccccccc) >> 2 );
695 Bits = ( (Bits & 0x55555555) << 1 ) | ( (Bits & 0xaaaaaaaa) >> 1 );
696 return Bits;
697}
