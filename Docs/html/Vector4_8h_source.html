968template <typename T>
969bool TVector4<T>::SerializeFromVector3(FName StructTag, FArchive& Ar)
970{
971 // Upgrade Vector3 - only set X/Y/Z. The W should already have been set to the property specific default and we don't want to trash it by forcing 0 or 1.
972 if(StructTag == NAME_Vector3f)
973 {
974 FVector3f AsVec;
975 Ar << AsVec;
976 X = (T)AsVec.X;
977 Y = (T)AsVec.Y;
978 Z = (T)AsVec.Z;
979 return true;
980 }
981 else if(StructTag == NAME_Vector || StructTag == NAME_Vector3d)
982 {
983 FVector3d AsVec; // Note: Vector relies on FVector3d serializer to handle float/double based on archive version.
984 Ar << AsVec;
985 X = (T)AsVec.X;
986 Y = (T)AsVec.Y;
987 Z = (T)AsVec.Z;
988 return true;
989 }
990 return false;
991}
